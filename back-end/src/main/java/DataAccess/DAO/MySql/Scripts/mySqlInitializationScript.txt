CREATE TABLE IF NOT EXISTS Users (
	Id VARCHAR(36) NOT NULL,
	Username VARCHAR(50) NOT NULL UNIQUE,
	PasswordHash VARCHAR(255) NOT NULL,
	PasswordSalt VARCHAR(24) NOT NULL,
	FirstName VARCHAR(25) NOT NULL,
	LastName VARCHAR(25) NOT NULL,
	Email VARCHAR(50),
	PhoneNumber VARCHAR(25),
	ImageUrl VARCHAR(50),
	PRIMARY KEY (Id)
);
CREATE TABLE IF NOT EXISTS ItemCategories (
	Id VARCHAR(36) NOT NULL,
	CategoryName VARCHAR(25) NOT NULL,
	OwnerId VARCHAR(36) NOT NULL,
	PRIMARY KEY (Id),
	FOREIGN KEY (OwnerId) REFERENCES Users(Id)
);
CREATE TABLE IF NOT EXISTS Items (
	Id VARCHAR(36) NOT NULL,
	Title VARCHAR(50) NOT NULL,
	Category VARCHAR(25) NOT NULL,
	DateCreated VARCHAR(30) NOT NULL,
	Available BOOL NOT NULL,
	OwnerId VARCHAR(36) NOT NULL,
	ImageUrl VARCHAR(50),
	Description VARCHAR(255),
	NumPlayers INT,
	TimeToPlayInMins INT,
	ReleaseYear INT,
	Genre VARCHAR(50),
	ItemFormat VARCHAR(25),
	Author VARCHAR(50),
	PRIMARY KEY (Id),
	FOREIGN KEY (OwnerId) REFERENCES Users(Id)
);
CREATE TABLE IF NOT EXISTS Friendships (
	Id VARCHAR(36) NOT NULL,
	UserIdA VARCHAR(36) NOT NULL,
	UserIdB VARCHAR(36) NOT NULL,
	PRIMARY KEY (Id),
	FOREIGN KEY (UserIdA) REFERENCES Users(Id),
	FOREIGN KEY (UserIdB) REFERENCES Users(Id)
);

CREATE TABLE IF NOT EXISTS AuthTokens (
	Id VARCHAR(36) NOT NULL,
	Token VARCHAR(36) NOT NULL,
	OwnerId VARCHAR(36) NOT NULL,
	PRIMARY KEY (Id),
	FOREIGN KEY (OwnerId) REFERENCES Users(Id)
);
